Option Explicit

Private Sub Worksheet_SelectionChange(ByVal Target As Range)
    Dim Password
    Dim rngPermessi As Range
    Dim LastRow As Integer
    Password = "Ej20082018*Excel"
    
    LastRow = 4
    While Cells(LastRow, 1) <> ""
        LastRow = LastRow + 1
    Wend
    LastRow = LastRow - 1
    
    Set rngPermessi = Range("AI4", Cells(LastRow, 49))
    
    If Not Intersect(Target, rngPermessi) Is Nothing Or Not Intersect(Target, Range("C52")) Is Nothing Then
        ActiveSheet.Unprotect Password:=Password
    Else
        ActiveSheet.Protect Password:=Password
    End If
End Sub
